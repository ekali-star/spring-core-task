package com.example.gymcrm;

import com.example.gymcrm.config.AppConfig;
import com.example.gymcrm.facade.GymFacade;
import com.example.gymcrm.model.*;
import org.springframework.context.ApplicationContext;
import org.springframework.context.annotation.AnnotationConfigApplicationContext;

import java.time.LocalDate;

public class GymCrmApplication {
    public static void main(String[] args) {
        ApplicationContext context =
                new AnnotationConfigApplicationContext(AppConfig.class);

        GymFacade facade = context.getBean(GymFacade.class);

        System.out.println("========================================");
        System.out.println("Gym CRM Application started successfully!");
        System.out.println("========================================");

        System.out.println("\n=== Testing Trainee Creation ===");
        Trainee newTrainee = new Trainee();
        newTrainee.setFirstName("Test");
        newTrainee.setLastName("User");
        newTrainee.setDateOfBirth(LocalDate.of(1995, 5, 15));
        newTrainee.setAddress("123 Test St");

        Trainee created = facade.createTrainee(newTrainee);
        System.out.println("Created trainee: " + created.getUsername());

        System.out.println("\n=== All Trainees ===");
        facade.getAllTrainees().forEach(t ->
                System.out.println(t.getUsername() + " - " + t.getFirstName() + " " + t.getLastName())
        );
    }
}
